89 ! - ierror: Integer for holding MPI error flag IDs (INTEGER).

92 SUBROUTINE parallel(dim1_len, dim2_len, matrix, min_val)
93 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
94 
95 USE mpi
96

98 
99 INTEGER :: dim1_len, &
