2 ! DECOMP_GRID. Defines a grid decomposition by dividing the grid as evenly as
3 ! possible along the second dimension (dim2), with the desired amount of space
4 ! needed to store the overlap regions (the regions that need to be communicated
5 ! between sub-grids). For example, this will divide an (8,9) grid among 3
6 ! processors with an overlap of 1 into three sub-grids of size (8,4), (8,5),
7 ! (8,4).
8 !
9 ! NOTES: - overlap is the number of rows a sub-grid needs to borrow from one of
10 ! its neighbors. For example, a first-order center difference approximation
11 ! of a second derivative needs the points i-1, i, i+1.
12 !
13 ! ARGUMENTS: - dim2_len_total: The size of dim2 of the grid (INTEGER).
14 ! - overlap: The overlap required for the numerical scheme (INTEGER).
15 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

66 ! PARALLEL. Handles the execution if there is more than one processor.
67 !
68 ! STRUCTURE: 1) Divide dim2_len as easily as possible, adding 1 where necessary,
69 ! e.g., splitting dim2_len = 9 among two processors, processor 0 needs 5 and
70 ! processor 1 needs 4.
71 !
72 ! VARIABLES: - dim2_len_total: The size of dim2 of the grid (INTEGER).
73 ! - overlap: The overlap required for the numerical scheme (INTEGER).
74 ! - proc_id: Processor ID (INTEGER).
75 ! - proc_count: Total number of processors (INTEGER).
76 ! - dim2_len: Size of dim2 of the sub-grid (INTEGER).
77 ! - ierror: Integer for holding MPI error flag IDs (INTEGER).
78 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
79 SUBROUTINE parallel(dim2_len_total, overlap, proc_id, proc_count)
80 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
81 
82 USE mpi
83 
84 IMPLICIT NONE
85 
86 INTEGER :: dim2_len_total, &
87 & overlap, &
88 & proc_id, &
89 & proc_count, &
90 & dim2_len, &
91 & ierror
92 
93 ! Divide dim2_len_total as evenly as possible, ignoring overlap.
94 dim2_len = dim2_len_total/proc_count
95 
96 ! Add 1 to dim2_len if the sub-grid would be big enough, see STRUCTURE for
97 ! example.
98 IF (proc_id .LT. modulo(dim2_len_total, proc_count)) THEN
99  dim2_len = dim2_len + 1
100 END IF
101 
102 ! Increase dim2_len for overlap.
103 ! Interior sub-grid.
104 IF (mod(proc_id, proc_count-1) .NE. 0) THEN
105  dim2_len = dim2_len + 2 * overlap
106 ! Boundary sub-grid.
107 ELSE IF (modulo(proc_id, proc_count-1) .EQ. 0) THEN
108  dim2_len = dim2_len + overlap
109 END IF
110 
111 dim2_len_total = dim2_len
112 
113 END SUBROUTINE parallel

125 ! - proc_count: Total number of processors (INTEGER).
126 ! - proc_err_flag - Flag is triggered if processor encounters an error
127 ! (LOGICAL).
128 ! - global_err_flag - Flag is triggered if any processor encouters an error
129 ! (LOGICAL).
130 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
131 SUBROUTINE error_handling(dim2_len_total, overlap, proc_id, proc_count)
132 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
133 
134 USE mpi
135 
136 IMPLICIT NONE
137 
138 INTEGER :: dim2_len_total, &
139 & overlap, &
140 & proc_id, &
141 & proc_count
142 LOGICAL :: proc_err_flag, &
143 & global_err_flag
144 
145 CALL mpi_barrier(mpi_comm_world, ierror)
146 
147 proc_err_flag = .false.
148 global_err_flag = .false.
149 
150 ! overlap too large. Only processor 0 needs to check this.
151 IF ((proc_id .EQ. 0) .AND. (overlap .GE. dim2_len_total/proc_count)) THEN
152  proc_err_flag = .true.
153  print *, 'EZ_PARALLEL: Issue with subroutine call DECOMP_GRID. overlap: ', &
154  & overlap, ' is greater than or equal to dim2_len ', &
155  & dim2_len_total/proc_count, ' of some processor.'
156  print *, 'Please decrease the overlap, decrease the number of processors, ', &
157  & 'or increase the size of the second dimension of the grid.'
158 END IF
159 
160 ! overlap < 1. Only processor 0 needs to check this.
161 IF ((proc_id .EQ. 0) .AND. (overlap .LT. 0)) THEN
162  proc_err_flag = .true.
163  print *, 'EZ_PARALLEL: Issue with subroutine call DECOMP_GRID. overlap: ', &
164  & overlap, ' is negative.'
165  print *, 'Please increase the overlap.'
166 END IF
167 
168 ! STOP all processors if any encounter an error.
169 CALL mpi_allreduce(proc_err_flag, global_err_flag, 1, mpi_logical, mpi_lor, &
170 & mpi_comm_world, ierror)
171 
172 IF (global_err_flag) THEN
173  CALL mpi_finalize(ierror)
174  stop
175 END IF
176 
177 CALL mpi_barrier(mpi_comm_world, ierror)
178 
179 END SUBROUTINE error_handling
