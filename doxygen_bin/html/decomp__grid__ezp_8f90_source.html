79  ! Store grid data in the grid.

81  grid_decomp%col_count_g = col_count

102  send_boundaries(1), ierror)
103  CALL mpi_type_create_subarray(2, (/row_count, col_count/), &
104  (/0, 0/), (/0,0/), mpi_order_fortran, mpi_double_precision, &

115  grid_decomp%RECV_BOUNDARIES = recv_boundaries

151  SUBROUTINE parallel_dble(row_count, col_count, overlap, decomp_id)
152 
153  USE mpi
154  USE ez_parallel_structs
155 
156  IMPLICIT NONE
157 
158  INTEGER, INTENT(IN) :: row_count
159  INTEGER, INTENT(INOUT) :: col_count
160  INTEGER, INTENT(IN) :: overlap
161  INTEGER, INTENT(IN) :: decomp_id
162  INTEGER :: row_count_s
164  INTEGER :: col_count_s
166  INTEGER :: SEND_BOUNDARIES(2)
168  INTEGER :: RECV_BOUNDARIES(2)
170  INTEGER :: ierror
171  INTEGER :: i
172  TYPE(grid_decomposition) :: grid_decomp
173 
174  ! Store grid data in the grid.
175  grid_decomp%row_count_g = row_count
176  grid_decomp%col_count_g = col_count
177  grid_decomp%overlap = overlap
178  grid_decomp%decomp_id = decomp_id
179 
180  ! Generate the horzontal-slab decomposition.
181  ALLOCATE(grid_decomp%row_decomp(proc_count))
182  ! Divide row_count as evenly as possible among all the processors, ignoring
183  ! overlap.
184  row_count_s = row_count / proc_count
185  DO i = 1, proc_count
186  IF (i .LE. modulo(row_count, proc_count)) THEN
187  grid_decomp%row_decomp(i) = row_count_s + 1
188  ELSE
189  grid_decomp%row_decomp(i) = row_count_s
190  END IF
191  END DO
192 
193  ! Generate the vertical-slab decomposition without overlap.
194  ALLOCATE(grid_decomp%col_decomp(proc_count))
195  ! Divide col_count as evenly as possible among all the processors, ignoring
196  ! overlap.
197  col_count_s = col_count/proc_count
198  DO i = 1, proc_count
199  IF (i .LE. modulo(col_count, proc_count)) THEN
200  grid_decomp%col_decomp(i) = col_count_s + 1
201  ELSE
202  grid_decomp%col_decomp(i) = col_count_s
203  END IF
204  END DO
205 
206  ! Generate the vertical-slab decomposition with overlap.
207  ALLOCATE(grid_decomp%col_decomp_ovlp(proc_count))
208  ! Different from without overlap only if overlap .NE. 0
209  IF (overlap .EQ. 0) THEN
210  grid_decomp%col_decomp_ovlp = grid_decomp%col_decomp
211  ELSE
212  ! All sub-grids with one neighbor get overlap additional columns.
213  grid_decomp%col_decomp_ovlp(1) = grid_decomp%col_decomp(1) + overlap
214  grid_decomp%col_decomp_ovlp(proc_count) = &
215  grid_decomp%col_decomp(proc_count) + overlap
216  ! All sub-grids with two neighbors get 2*overlap additional columns.
217  grid_decomp%col_decomp_ovlp(2:proc_count-1) = &
218  grid_decomp%col_decomp(2:proc_count-1) + 2*overlap
219  END IF
220  col_count = grid_decomp%col_decomp_ovlp(proc_id+1) ! Overwrite col_count
221  ! so user's previous code works.
222 
223  ! Create the SEND BOUNDARY datatypes, to be committed later.
224  CALL mpi_type_create_subarray(2, (/row_count, col_count/), &
225  (/row_count, overlap/), (/0,overlap/), mpi_order_fortran, &
226  mpi_double_precision, send_boundaries(1), ierror)
227  CALL mpi_type_create_subarray(2, (/row_count, col_count/), &
228  (/row_count, overlap/), (/0,col_count-1-(2*overlap-1)/), &
229  mpi_order_fortran, mpi_double_precision, send_boundaries(2), ierror)
230  grid_decomp%SEND_BOUNDARIES = send_boundaries
231 
232  ! Create the RECV BOUNDARY datatypes, to be committed later.
233  CALL mpi_type_create_subarray(2, (/row_count, col_count/), &
234  (/row_count, overlap/), (/0,0/), mpi_order_fortran, &
235  mpi_double_precision, recv_boundaries(1), ierror)
236  CALL mpi_type_create_subarray(2, (/row_count, col_count/), &
237  (/row_count, overlap/), (/0,col_count-1-(overlap-1)/), mpi_order_fortran, &
238  mpi_double_precision, recv_boundaries(2), ierror)
239  grid_decomp%RECV_BOUNDARIES = recv_boundaries
240 
241  ! Put grid_decomp into the list of unique grid_decomps
242  grid_decomps(decomp_id) = grid_decomp
243 
244  ! Commit the MPI derived datatypes
245  CALL mpi_type_commit(grid_decomps(decomp_id)%SEND_BOUNDARIES(1), ierror)
246  CALL mpi_type_commit(grid_decomps(decomp_id)%SEND_BOUNDARIES(2), ierror)
247  CALL mpi_type_commit(grid_decomps(decomp_id)%RECV_BOUNDARIES(1), ierror)
248  CALL mpi_type_commit(grid_decomps(decomp_id)%RECV_BOUNDARIES(2), ierror)
249 
250  ! Free allocated arrays
251  DEALLOCATE(grid_decomp%row_decomp)
252  DEALLOCATE(grid_decomp%col_decomp)
253  DEALLOCATE(grid_decomp%col_decomp_ovlp)
254 
255  END SUBROUTINE parallel_dble

293  SUBROUTINE error_handling_dble(row_count, col_count, overlap, decomp_id)
294 
295  USE mpi
296  USE ez_parallel_structs
297 
298  IMPLICIT NONE
299 
300  INTEGER, INTENT(IN) :: row_count
301  INTEGER, INTENT(IN) :: col_count
302  INTEGER, INTENT(IN) :: overlap
303  INTEGER, INTENT(IN) :: decomp_id
304  INTEGER :: error_code
305  INTEGER :: ierror
306  LOGICAL :: error_flag
307 
308  error_flag = .false.
309 
310  ! Check if row_count is not positive.
311  IF (row_count .LT. 1) THEN
312  IF (proc_id .EQ. 0) THEN
313  print *, "ERROR: Non-positive number of rows in grid with ", &
314  " decomposition ID ", decomp_id, ". Number of rows: ", &
315  row_count, "."
316  END IF
317  error_flag = .true.
318  END IF
319 
320  ! Check if col_count is not positive.
321  IF (col_count .LT. 1) THEN
322  IF (proc_id .EQ. 0) THEN
323  print *, "ERROR: Non-positive number of columns in grid with ", &
324  " decomposition ID ", decomp_id, ". Number of columns: ", &
325  col_count, "."
326  END IF
327  error_flag = .true.
328  END IF
329 
330  ! Check if col_count is too small to be divided among processors, excluding overlap.
331  IF (col_count .LT. proc_count) THEN
332  IF (proc_id .EQ. 0) THEN
333  print *, "ERROR: Number of columns in grid with decomposition ID ", &
334  decomp_id, " fewer than number of processors. Number of columns: ", &
335  col_count, ". Number of processors: ", proc_count, "."
336  END IF
337  error_flag = .true.
338  END IF
339 
340  ! Check if col_count is too small to be divided among processors, including overlap.
341  IF (col_count .LT. overlap*proc_count) THEN
342  IF (proc_id .EQ. 0) THEN
343  print *, "ERROR: Number of columns too small to decompose grid ", &
344  "without sub-grid boundaries overlapping for grid with ", &
345  "decomposition ID ", decomp_id, ". Number of columns: ", &
346  col_count, ". Overlap: ", overlap , "."
347  END IF
348  error_flag = .true.
349  END IF
350 
351  ! Check if overlap is negative.
352  IF (overlap .LT. 0) THEN
353  IF (proc_id .EQ. 0) THEN
354  print *, "ERROR: Negative overlap in grid with decomposition ID ", &
355  decomp_id, ". overlap: ", overlap, "."
356  END IF
357  error_flag = .true.
358  END IF
359 
360  ! Check if decomp_id outside of (1, ..., decomp_count)
361  IF ((decomp_id .LT. 1) .OR. (decomp_id .GT. decomp_count)) THEN
362  IF (proc_id .EQ. 0) THEN
363  print *, "ERROR: Decomposition ID in grid with decomposition ID ", &
364  decomp_id, " is outside of 1, ..., decomp_count. decomp_count: ", &
365  decomp_count, "."
366  END IF
367  error_flag = .true.
368  END IF
369 
370  IF (error_flag) THEN
371  CALL mpi_abort(mpi_comm_world, error_code, ierror)
372  END IF
373 
374  END SUBROUTINE error_handling_dble
375  !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
376 
377 END SUBROUTINE decomp_grid_dble

714  END IF

734  END IF

736  END IF

739  IF ((decomp_id .LT. 1) .OR. (decomp_id .GT. decomp_count)) THEN
740  IF (proc_id .EQ. 0) THEN
741  print *, "ERROR: Decomposition ID in grid with decomposition ID ", &
742  decomp_id, " is outside of 1, ..., decomp_count. decomp_count: ", &
743  decomp_count, "."
744  END IF
745  error_flag = .true.
746  END IF
747 
748  IF (error_flag) THEN
749  CALL mpi_abort(mpi_comm_world, error_code, ierror)
750  END IF
