2 ! SPECTRAL_DIM2_DERIVATIVE. Produces the matrix used in calculating the
3 ! derivative of the numerical solution along the second dimension (dim2) of the
4 ! grid.
5 !
6 ! ARGUMENTS: - dim1_len, dim2_len: The shape of the sub-grid (INTEGER).
7 ! - overlap: The overlap required for the numerical scheme (INTEGER).
8 ! - spec_dim2_deriv: The array that will store the matrix used in calculating
9 ! the derivative of the numerical solution along the second dimension of the
10 ! grid (DOUBLE COMPLEX, DIMENSION(dim1_len, dim2_len)).
11 ! - order: The order of the derviate desired (INTEGER).
12 !
13 ! NOTES: - When calculating the dim2 wavenumbers, we use the convention
14 ! [0, ..., dim2_len/2, -dim2_len/2 + 1, ... -1], with FLOOR(dim2_len/2), as
15 ! is assumed in Fortran integer division.
16 !
17 ! Written By: Jason Turner
18 ! Last Updated: January 12, 2020

58 ! SERIAL. Handles the execution if there is only one processor.
59 !
60 ! STRUCTURE: 1) Calculate the dim2 wavenumbers.
61 ! 2) If dim2_len even and order odd, zero out the dim2_len/2 wavenumber.
62 ! 3) Populate spec_dim2_deriv array with (sqrt(-1)*dim2 wavenumber)**order.
63 !
64 ! VARIABLES: - dim1_len, dim2_len: The shape of the sub-grid (INTEGER).
65 ! - overlap: The overlap required for the numerical scheme (INTEGER).
66 ! - spec_dim2_deriv: The array that will store the matrix used in calculating
67 ! the derivative of the numerical solution along the second dimension of the
68 ! grid (DOUBLE COMPLEX, DIMENSION(dim1_len, dim2_len)).
69 ! - order: The order of the derviate desired (INTEGER).
70 ! - i: Counting index used in DO loops (INTEGER).
71 ! - dim2_wavenums: Array used to store the dim2 wavenumbers (DOUBLE PRECISION,
72 ! DIMENSION(dim2_len)).
73 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
74 SUBROUTINE serial(dim1_len, dim2_len, overlap, spec_dim2_deriv, order)
75 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
76 
77 USE mpi
78 
79 IMPLICIT NONE
80 
81 INTEGER :: dim1_len, &
82 & dim2_len, &
83 & overlap, &
84 & order, &
85 & i
86 DOUBLE PRECISION, DIMENSION(dim2_len) :: dim2_wavenums
87 DOUBLE COMPLEX, DIMENSION(dim1_len, dim2_len) :: spec_dim2_deriv
88 
89 ! Calculate dim2 wavenumbers.
90 dim2_wavenums = 0.0
91 dim2_wavenums(2) = 1.0
92 dim2_wavenums(dim2_len) = -1.0
93 DO i = 1, dim2_len/2 - 1
94  dim2_wavenums(dim2_len - i) = dble(-i - 1)
95  dim2_wavenums(i + 2) = dble(i + 1)
96 END DO
97 ! If dim2_len even, we have to zero out highest wavenumber for odd-order
98 ! derivatives.
99 IF ((mod(dim2_len, 2) .EQ. 0) .AND. (mod(order, 2) .EQ. 1)) THEN
100  dim2_wavenums(dim2_len/2 + 1) = 0.0
101 END IF
102 
103 ! Populate spec_dim2_deriv array.
104 DO i = 1, dim2_len
105  spec_dim2_deriv(:,i) = (dcmplx(0.0, dim2_wavenums(i)))**(dble(order))
106 END DO
107 
108 END SUBROUTINE serial

112 ! PARALLEL. Handles the execution if there is more than one processor.
113 !
114 ! STRUCTURE: 1) Recover the size of the (original) grid.
115 ! 2) Find where the processor's sub-grid (referred to simply as "sub-grid"
116 ! henceforth) is in the grid.
117 ! 3) Calculate the dim2 wavenumbers for the sub-grid. If dim2_len_total even
118 ! and order odd, zero out the dim2_len/2 wavenumber.
119 ! 4) Populate spec_dim2_deriv array with (sqrt(-1)*dim2 wavenumber)**order.
120 !
121 ! VARIABLES: - dim1_len, dim2_len: The shape of the sub-grid (INTEGER).
122 ! - overlap: The overlap required for the numerical scheme (INTEGER).
123 ! - spec_dim2_deriv: The array that will store the matrix used in calculating
124 ! the derivative of the numerical solution along the second dimension of the
125 ! grid (DOUBLE COMPLEX, DIMENSION(dim1_len, dim2_len)).
126 ! - order: The order of the derviate desired (INTEGER).
127 ! - proc_id: Processor ID (INTEGER).
128 ! - proc_count: Total number of processors (INTEGER).
129 ! - dim2_len_raw: Dim2 length of the interior of the sub-grid (INTEGER).
130 ! - dim2_len_total: Dim2 length of the grid (INTEGER).
131 ! - dim2_grid_ind_low, dim2_grid_ind_high: The dim2 indices of the sub-grid in
132 ! the grid (if the sub-grid is indexed locally, these are the global dim2
133 ! indices) (INTEGER).
134 ! - dim2_len_pre: Dim2 length of all processors with lesser ID, i.e., the global
135 ! dim2 index for the non-overlap part of the sub-grid (INTEGER).
136 ! - ierror: Integer for holding MPI error flag IDs (INTEGER).
137 ! - i: Counting index used in DO loops (INTEGER).
138 ! - dim2_len_list: Array of the dim2_len for all processors (INTEGER,
139 ! DIMENSION(proc_count)).
140 ! - dim2_wavenums: Array used to store the dim2 wavenumbers (DOUBLE PRECISION,
141 ! DIMENSION(dim2_len)).
142 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
143 SUBROUTINE parallel(dim1_len, dim2_len, overlap, spec_dim2_deriv, order, &
144 & proc_id, proc_count)
145 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
146 
147 USE mpi
148 
149 IMPLICIT NONE
150 
151 INTEGER :: dim1_len, &
152 & dim2_len, &
153 & overlap, &
154 & order, &
155 & dim2_len_raw, &
156 & dim2_len_total, &
157 & dim2_grid_ind_low, &
158 & dim2_grid_ind_high, &
159 & dim2_len_pre, &
160 & proc_id, &
161 & proc_count, &
162 & ierror, &
163 & i
164 INTEGER, DIMENSION(proc_count) :: dim2_len_list
165 DOUBLE PRECISION, DIMENSION(dim2_len) :: dim2_wavenums
166 DOUBLE COMPLEX, DIMENSION(dim1_len, dim2_len) :: spec_dim2_deriv
167 
168 ! Calculate dim2 of sub-grid to send along.
169 IF (mod(proc_id, proc_count-1) .EQ. 0) THEN
170  ! First or last processor processor.
171  dim2_len_raw = dim2_len - overlap
172 ELSE
173  ! Middle processor.
174  dim2_len_raw = dim2_len - 2 * overlap
175 END IF
176 
177 CALL mpi_allreduce(dim2_len_raw, dim2_len_total, 1, mpi_integer, mpi_sum, &
178 & mpi_comm_world, ierror)
179 
180 ! Calculate dim2_len for each sub-grid, to avoid communication between processors.
181  ! Divide dim2_len_total as evenly as possible.
182 dim2_len_raw = dim2_len_total/proc_count
183  ! Now check if sub-grid of a given processor needs to take additional dim2_len,
184  ! i.e., dim2_len_total % proc_count != 0.
185 DO i = 1, proc_count
186  IF ((i-1) .LT. modulo(dim2_len_total, proc_count)) THEN
187  dim2_len_list(i) = dim2_len_raw + 1
188  ! The indexing is weird here since processors start at index 0 but arrays
189  ! start at index 1. Thus, entry ii of this corresponds to processor (i-1).
190  ELSE
191  dim2_len_list(i) = dim2_len_raw
192  END IF
193 END DO
194 
195 ! Calculate the dim2 grid indices of the sub-grid (the global coordinates
196 ! of the sub-grid).
197 dim2_len_pre = sum(dim2_len_list(1:proc_id))
198 IF (proc_id .EQ. 0) THEN
199  dim2_grid_ind_low = dim2_len_pre + 1
200  dim2_grid_ind_high = dim2_len_pre + dim2_len_raw + overlap
201 ELSE IF (proc_id .EQ. proc_count - 1) THEN
202  dim2_grid_ind_low = dim2_len_pre + 1 - overlap
203  dim2_grid_ind_high = dim2_len_pre + dim2_len_raw
204 ELSE
205  dim2_grid_ind_low = dim2_len_pre + 1 - overlap
206  dim2_grid_ind_high = dim2_len_pre + dim2_len_raw + overlap
207 END IF
208 
209 
210 ! Calculate dim2 wavenumbers.
211 ! If dim2_len_total even, we have to zero out highest wavenumber for odd-order
212 ! derivatives.
213 IF (mod(dim2_len_total, 2) .EQ. 0) THEN
214  DO i = 1, dim2_len
215  ! Non-negative wavenumbers.
216  IF (i+dim2_grid_ind_low-1 .LE. dim2_len_total/2+1) THEN
217  dim2_wavenums(i) = dble(i+dim2_grid_ind_low-2)
218  ! Negative wavenumbers.
219  ELSE IF (i+dim2_grid_ind_low-1 .GT. dim2_len_total/2+1) THEN
220  dim2_wavenums(i) = dble(i-dim2_len_total+dim2_grid_ind_low-2)
221  ELSE
222  IF (mod(order, 2) .EQ. 1) THEN
223  dim2_wavenums(i) = 0.0
224  END IF
225  END IF
226  END DO
227 ELSE
228  DO i = 1, dim2_len
229  ! Non-negative wavenumbers.
230  IF (i+dim2_grid_ind_low-1 .LE. dim2_len_total/2+1) THEN
231  dim2_wavenums(i) = dble(i+dim2_grid_ind_low-2)
232  ! Negative wavenumbers.
233  ELSE IF (i+dim2_grid_ind_low-1 .GT. dim2_len_total/2+1) THEN
234  dim2_wavenums(i) = dble(i-dim2_len_total+dim2_grid_ind_low-2)
235  END IF
236  END DO
237 END IF
238 
239 ! Populate spec_dim2_deriv array with (sqrt(-1)*dim2 wavenumber)**order.
240 DO i = 1, dim2_len
241  spec_dim2_deriv(:,i) = (dcmplx(0.0, dim2_wavenums(i)))**(dble(order))
242 END DO
243 
244 END SUBROUTINE parallel

248 ! ERROR_HANDLING. Checks for input error and other possible complications.
249 !
250 ! ERRORS TO CHECK: - order < 1.
251 !
252 ! VARIABLES: - order: The order of the derviate desired (INTEGER).
253 ! - proc_id: Processor ID (INTEGER).
254 ! - proc_err_flag - Flag is triggered if processor encounters an error
255 ! (LOGICAL).
256 ! - global_err_flag - Flag is triggered if any processor encouters an error
257 ! (LOGICAL).
258 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
259 SUBROUTINE error_handling(order, proc_id)
260 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
261 
262 USE mpi
263 
264 IMPLICIT NONE
265 
266 INTEGER :: order, &
267 & proc_id
268 LOGICAL :: proc_err_flag, &
269 & global_err_flag
270 
271 CALL mpi_barrier(mpi_comm_world, ierror)
272 
273 proc_err_flag = .false.
274 global_err_flag = .false.
275 
276 ! order < 1.
277 IF (order .LT. 1) THEN
278  proc_err_flag = .true.
279  print *, 'EZ_PARALLEL: Issue with processor ', proc_id, ' in subroutine ', &
280  & 'call SPECTRAL_DIM2_DERIVATIVE_DBLE_CMPLX. order: ', order, ' is less ', &
281  & 'than 1. Please increase the order.'
282 END IF
283 
284 ! STOP all processors if any encounter an error.
285 CALL mpi_allreduce(proc_err_flag, global_err_flag, 1, mpi_logical, mpi_lor, &
286 & mpi_comm_world, ierror)
287 
288 IF (global_err_flag) THEN
289  CALL mpi_finalize(ierror)
290  stop
291 END IF
292 
293 CALL mpi_barrier(mpi_comm_world, ierror)
294 
295 END SUBROUTINE error_handling
